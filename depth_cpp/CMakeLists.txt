message(STATUS "OpenCV library status:")
message(STATUS "    config: ${OpenCV_DIR}")
message(STATUS "    version: ${OpenCV_VERSION}")
message(STATUS "    libraries: ${OpenCV_LIBS}")
message(STATUS "    include path: ${OpenCV_INCLUDE_DIRS}")

cmake_minimum_required(VERSION 3.5)

project(disparity-with-unwrap)

find_package(vpi 1.0 REQUIRED)
find_package(OpenCV REQUIRED)

add_executable(${PROJECT_NAME} main.cpp)
target_link_libraries(${PROJECT_NAME} vpi opencv_imgproc opencv_core opencv_videoio opencv_highgui opencv_calib3d)

if(OpenCV_VERSION VERSION_LESS 3)
    target_link_libraries(${PROJECT_NAME} opencv_highgui)
else()
    target_link_libraries(${PROJECT_NAME} opencv_imgcodecs)
endif()
